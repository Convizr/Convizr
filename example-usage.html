Integration Code

Here's how to integrate the extensions in your application:

<script type="module"> // Import the extension system import { extensionRegistry, registerExtension, renderTraceWithExtensions } from 'https://convizr.github.io/Convizr/ConvizrExtensionRegistry.js'; import { OnboardingFormExtension } from 'https://convizr.github.io/Convizr/ConvizrExtensions_V0.1.js'; // Initialize extensions function initializeExtensions(voiceflowClient) { extensionRegistry.setVoiceflowClient(voiceflowClient); registerExtension(OnboardingFormExtension); } // Handle traces with extensions async function renderTraces(traces, container) { for (const trace of traces) { const extensionHandled = await renderTraceWithExtensions(trace, container); if (!extensionHandled) { // Handle with your default trace rendering logic console.log('Default trace handling for:', trace.type); } } } // Initialize Voiceflow (replace with your actual credentials) const voiceflowClient = new Voiceflow({ versionID: 'your-version-id', apiKey: 'your-api-key' }); // Setup and start initializeExtensions(voiceflowClient); voiceflowClient.onMessage((message) => { const container = document.getElementById('chat-container'); renderTraces(message.traces || [], container); }); voiceflowClient.start(); </script>
